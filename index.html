updateBooking() { const fullName = document.getElementById('fullName').value.trim(); const email = document.getElementById('email').value.trim(); const phone = document.getElementById('phone').value.trim(); const date = document.getElementById('courseDate').value; const time = document.getElementById('selectedTime').value; if (!fullName || !email || !date || !time) { showNotification('❌ Заполните все обязательные поля', 'error'); return; } const bookingIndex = this.bookings.findIndex(b => b.id === this.editingBookingId); if (bookingIndex !== -1) { this.bookings[bookingIndex] = { ...this.bookings[bookingIndex], fullName: fullName, email: email, phone: phone, date: date, time: time }; this.saveBookings(); this.renderBookings(); this.updateStats(); this.resetForm(); showNotification('✅ Запись обновлена!', 'success'); } } cancelBooking(id) { if (confirm('Вы уверены, что хотите отменить эту запись?')) { const booking = this.bookings.find(b => b.id === id); if (booking) { booking.status = 'cancelled'; this.saveBookings(); this.renderBookings(); this.updateStats(); showNotification('🚫 Запись отменена', 'info'); } } } restoreBooking(id) { const booking = this.bookings.find(b => b.id === id); if (booking) { booking.status = 'active'; this.saveBookings(); this.renderBookings(); this.updateStats(); showNotification('✅ Запись восстановлена', 'success'); } } deleteBooking(id) { if (confirm('Вы уверены, что хотите удалить эту запись безвозвратно?')) { this.bookings = this.bookings.filter(b => b.id !== id); this.saveBookings(); this.renderBookings(); this.updateStats(); if (this.editingBookingId === id) { this.resetForm(); } showNotification('🗑️ Запись удалена', 'info'); } } editBooking(id) { const booking = this.bookings.find(b => b.id === id); if (booking) { this.isEditing = true; this.editingBookingId = id; document.getElementById('fullName').value = booking.fullName; document.getElementById('email').value = booking.email; document.getElementById('phone').value = booking.phone || ''; document.getElementById('courseDate').value = booking.date; document.getElementById('selectedTime').value = booking.time; this.selectedTime = booking.time; document.getElementById('submitBtn').textContent = '💾 Сохранить изменения'; document.getElementById('editingIndicator').classList.add('active'); document.getElementById('editingId').textContent = id; this.renderTimeSlots('timeSlots', 'courseDate', 'selectedTime'); setTimeout(() => { const slots = document.querySelectorAll('#timeSlots .time-slot'); slots.forEach(slot => { if (slot.textContent.includes(booking.time)) { slot.classList.add('selected'); } }); }, 100); window.scrollTo({ top: 0, behavior: 'smooth' }); } } resetForm() { this.isEditing = false; this.editingBookingId = null; this.selectedTime = null; document.getElementById('bookingForm').reset(); document.getElementById('selectedTime').value = ''; document.getElementById('submitBtn').textContent = '✅ Записаться на курс'; document.getElementById('editingIndicator').classList.remove('active'); this.renderTimeSlots('timeSlots', 'courseDate', 'selectedTime'); } getActiveBookings() { return this.bookings.filter(b => b.status === 'active'); } getCancelledBookings() { return this.bookings.filter(b => b.status === 'cancelled'); } renderBookings() { const tbody = document.getElementById('bookingsTableBody'); const searchText = document.getElementById('searchInput').value.toLowerCase(); const filterDate = document.getElementById('filterDate').value; const filterStatus = document.getElementById('filterStatus').value; let filteredBookings = [...this.bookings]; if (searchText) { filteredBookings = filteredBookings.filter(b => b.fullName.toLowerCase().includes(searchText) || b.email.toLowerCase().includes(searchText) || b.phone.includes(searchText) ); } if (filterDate) { filteredBookings = filteredBookings.filter(b => b.date === filterDate); } if (filterStatus === 'active') { filteredBookings = filteredBookings.filter(b => b.status === 'active'); } else if (filterStatus === 'cancelled') { filteredBookings = filteredBookings.filter(b => b.status === 'cancelled'); } filteredBookings.sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt)); if (filteredBookings.length === 0) { tbody.innerHTML = 'Нет записей'; return; } tbody.innerHTML = filteredBookings.map((b, index) => ` ${index + 1} ${b.fullName} ${b.email} ${b.phone || '-'} ${new Date(b.date).toLocaleDateString('ru-RU')} ${b.time} ${b.status === 'active' ? 'Активна' : 'Отменена'}
${b.status === 'active' ? ` ` : ` `}
`).join(''); } filterBookings() { this.renderBookings(); } updateStats() { const totalBookings = this.getActiveBookings().length; const today = new Date().toISOString().split('T')[0]; const todayBookings = this.getActiveBookings().filter(b => b.date === today).length; let availableSlots = 0; const tomorrow = new Date(); tomorrow.setDate(tomorrow.getDate() + 1); const tomorrowStr = tomorrow.toISOString().split('T')[0]; this.timeSlots.forEach(slot => { const availability = this.getTimeSlotAvailability(slot.time, tomorrowStr); availableSlots += availability.available; }); document.getElementById('totalBookings').textContent = totalBookings; document.getElementById('todayBookings').textContent = todayBookings; document.getElementById('availableSlots').textContent = availableSlots; } validateEmail(email) { const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; return re.test(email); } }
